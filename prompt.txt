CONTEXTO DEL SISTEMA (LEE ESTO PRIMERO)
El problema: por qué no leemos Tibia directamente
Tibia usa BattleEye (anti-cheat). Si intentas capturar la ventana de Tibia con métodos
normales como PrintWindow de Win32, BattleEye lo detecta y devuelve una imagen completamente
negra. También bloquea la lectura directa de memoria del proceso.
La solución: usar OBS como intermediario
El flujo completo es este:
TIBIA (client.exe con BattleEye)
    ↓  renderiza con GPU/DirectX
OBS Studio → fuente "Captura de juego" (Game Capture)
    ↓  OBS hookea DirectX a nivel de driver (BattleEye lo permite porque es legítimo)
Proyector de OBS → ventana "Proyector - Fuente: Captura de juego"
    ↓  esta ventana proyecta el frame del juego y SÍ es capturable
Tu script Python con `mss`
    ↓  captura esa región de pantalla (screenshot de coordenadas)
Análisis de píxeles → detecta barra de HP/Mana
    ↓
Envía tecla directamente a Tibia via PostMessage Win32
Por qué mss funciona y PrintWindow no

PrintWindow: le pide a la ventana que se "redibuje" en un bitmap → falla con ventanas GPU
mss: captura directamente el framebuffer (lo que está visible en pantalla) → siempre funciona
El Proyector de OBS DEBE estar visible en pantalla (puede estar detrás de Tibia, pero NO minimizado)

def find_tibia_windows():
    results = []
    def callback(hwnd, _):
        if not win32gui.IsWindowVisible(hwnd):
            return
        title = win32gui.GetWindowText(hwnd)
        # Buscar "tibia" en el título (el nombre del personaje va después del guión)
        if "tibia" in title.lower():
            rect = win32gui.GetWindowRect(hwnd)
            w = rect[2] - rect[0]
            h = rect[3] - rect[1]
            if w > 300:  # filtrar ventanas muy pequeñas
                results.append({
                    "hwnd": hwnd,
                    "title": title,
                    "width": w,
                    "height": h
                })
    win32gui.EnumWindows(callback, None)
    return results

    results = []
    def callback(hwnd, _):
        if not win32gui.IsWindowVisible(hwnd):
            return
        title = win32gui.GetWindowText(hwnd)
        keywords = [
            "Proyector - Fuente",       # OBS en español
            "Projector - Source",        # OBS en inglés
            "Captura de juego",          # nombre de la fuente en español
            "Game Capture",              # nombre de la fuente en inglés
        ]
        if any(k.lower() in title.lower() for k in keywords):
            rect = win32gui.GetWindowRect(hwnd)
            x, y, x2, y2 = rect
            w, h = x2 - x, y2 - y
            if w > 200 and h > 200:
                results.append({"hwnd": hwnd, "title": title, 
                                 "x": x, "y": y, "w": w, "h": h})
    win32gui.EnumWindows(callback, None)
    # IMPORTANTE: Preferir "Fuente" sobre "Vista Previa" - la vista previa puede estar negra
    for r in results:
        if "fuente" in r["title"].lower() or "source" in r["title"].lower():
            return r
    return results[0] if results else None

pythonimport mss
import numpy as np
import cv2

def capture_projector(projector_info):
    """
    projector_info = {"x": 0, "y": 0, "w": 1382, "h": 744, ...}
    
    IMPORTANTE: mss captura coordenadas ABSOLUTAS de pantalla
    El proyector tiene bordes de Windows:
    - Borde lateral: ~8px
    - Barra de título: ~30px  
    - Borde inferior: ~8px
    """
    px = projector_info["x"]
    py = projector_info["y"]
    pw = projector_info["w"]
    ph = projector_info["h"]
    
    BORDER = 8
    TITLEBAR = 30
    
    monitor = {
        "left":   px + BORDER,
        "top":    py + TITLEBAR,    # saltar la barra de título
        "width":  pw - BORDER * 2,
        "height": ph - TITLEBAR - BORDER
    }
    
    with mss.mss() as sct:
        screenshot = sct.grab(monitor)
        img = np.array(screenshot)
        img = cv2.cvtColor(img, cv2.COLOR_BGRA2BGR)
    
    return img  # imagen BGR de numpy, lista para OpenCV

LAS BARRAS DE HP Y MANA: COLORES Y DETECCIÓN
Sistema de colores HSV
OpenCV usa HSV (Hue=Tono, Saturation=Saturación, Value=Brillo) en lugar de RGB porque
es más fácil definir rangos de color reales. En OpenCV: H va de 0-180, S y V van de 0-255.
         ROJO          AMARILLO      VERDE         AZUL
HSV:   0° y 180°         30°          60°           120°
       ────────────────────────────────────────────────
       |         |         |         |         |
       0        30        60        90        120       180

Algoritmo completo de detección:
pythondef detect_hp_mana(img):
    """
    Detecta el porcentaje de HP y Mana de la imagen capturada del proyector.
    
    Retorna: (hp_percent, mp_percent) donde cada uno es float entre 0.0 y 1.0
    Retorna None si no puede detectar una barra.
    """
    if img is None or img.size == 0:
        return None, None
    
    h, w = img.shape[:2]
    
    # PASO 1: Recortar solo la franja de las barras (top 10% de la imagen)
    scan_height = max(60, int(h * 0.10))
    region = img[0:scan_height, :]
    
    # PASO 2: Convertir a HSV
    hsv = cv2.cvtColor(region, cv2.COLOR_BGR2HSV)
    
    # PASO 3: Crear máscaras para HP (todos los colores posibles)
    mask_verde    = cv2.inRange(hsv, np.array([35, 60, 60]),   np.array([90, 255, 255]))
    mask_amarillo = cv2.inRange(hsv, np.array([20, 100, 100]), np.array([35, 255, 255]))
    mask_rojo1    = cv2.inRange(hsv, np.array([0, 100, 100]),  np.array([10, 255, 255]))
    mask_rojo2    = cv2.inRange(hsv, np.array([170, 100, 100]),np.array([180, 255, 255]))
    
    # Combinar todas las máscaras de HP en una sola
    mask_hp = cv2.bitwise_or(mask_verde, mask_amarillo)
    mask_hp = cv2.bitwise_or(mask_hp, mask_rojo1)
    mask_hp = cv2.bitwise_or(mask_hp, mask_rojo2)
    
    # Máscara para Mana (azul)
    mask_mp = cv2.inRange(hsv, np.array([95, 60, 60]), np.array([135, 255, 255]))
    
    # PASO 4: Calcular porcentajes
    # El ancho esperado de la barra al 100% es ~43% del ancho del proyector
    expected_full_width = w * 0.43
    
    hp_pct = calculate_bar_percent(mask_hp, expected_full_width)
    mp_pct = calculate_bar_percent(mask_mp, expected_full_width)
    
    return hp_pct, mp_pct


def calculate_bar_percent(mask, expected_full_width):
    """
    Dado un mask binario de la región de barras,
    calcula qué porcentaje del ancho máximo está llena la barra.
    """
    # Contar píxeles de color en cada fila
    row_sums = np.sum(mask > 0, axis=1)
    
    # Si no hay suficientes píxeles, no hay barra
    if row_sums.max() < 8:
        return None
    
    # La fila con más píxeles = la barra principal
    best_row_idx = np.argmax(row_sums)
    best_row = mask[best_row_idx]
    
    # Encontrar columnas con píxeles
    nonzero_cols = np.where(best_row > 0)[0]
    if len(nonzero_cols) < 5:
        return None
    
    # Encontrar el bloque continuo más largo (la barra real, no ruido)
    bar_length = find_longest_continuous_block(nonzero_cols, gap_tolerance=3)
    
    # Calcular porcentaje
    percent = min(bar_length / expected_full_width, 1.0)
    return round(percent, 3)


def find_longest_continuous_block(indices, gap_tolerance=3):
    """Encuentra el bloque continuo más largo, tolerando pequeños gaps."""
    if len(indices) == 0:
        return 0
    best = curr = 1
    for i in range(1, len(indices)):
        if indices[i] - indices[i-1] <= gap_tolerance:
            curr += 1
        else:
            best = max(best, curr)
            curr = 1
    return max(best, curr)

    "F4": win32con.VK_F4,   "F5": win32con.VK_F5,   "F6": win32con.VK_F6,

    "F10": win32con.VK_F10, "F11": win32con.VK_F11, "F12": win32con.VK_F12,
}

def send_key_to_window(hwnd, key_name):
    """
    Envía una tecla directamente a una ventana usando PostMessage.
    NO requiere que la ventana esté en foco/primer plano.
    
    hwnd: handle de la ventana de Tibia
    key_name: "F1", "F2", ..., "F12"
    """
    vk = VK_MAP.get(key_name.upper())
    if vk is None:
        print(f"ERROR: Tecla '{key_name}' no reconocida")
        return False
    
    scan_code = win32api.MapVirtualKey(vk, 0)
    lParam_down = (scan_code << 16) | 1
    lParam_up   = (scan_code << 16) | 0xC0000001
    
    try:
        win32api.PostMessage(hwnd, win32con.WM_KEYDOWN, vk, lParam_down)
        time.sleep(0.05)  # pequeña pausa entre down y up
        win32api.PostMessage(hwnd, win32con.WM_KEYUP,   vk, lParam_up)
        return True
    except Exception as e:
        print(f"ERROR enviando tecla: {e}")
        return False

Calibración automática de barras:
pythondef auto_calibrate(img):
    """
    Escanea toda la franja superior de la imagen y encuentra
    automáticamente en qué fila (y) están las barras de HP y Mana.
    
    Útil cuando el usuario cambia la resolución del proyector.
    
    Retorna: {"hp_row": int, "mp_row": int, "bar_width": int}
    """
    h, w = img.shape[:2]
    scan_h = int(h * 0.15)
    region = img[0:scan_h, :]
    hsv = cv2.cvtColor(region, cv2.COLOR_BGR2HSV)
    
    results = {"hp_row": None, "mp_row": None, "bar_max_width": 0}
    
    for y in range(scan_h):
        row_hsv = hsv[y:y+1, :]
        
        # Contar píxeles de cada color
        green_px  = cv2.countNonZero(cv2.inRange(row_hsv, [35,60,60],   [90,255,255]))
        yellow_px = cv2.countNonZero(cv2.inRange(row_hsv, [20,100,100], [35,255,255]))
        red_px    = cv2.countNonZero(cv2.inRange(row_hsv, [0,100,100],  [10,255,255]))
        blue_px   = cv2.countNonZero(cv2.inRange(row_hsv, [95,60,60],   [135,255,255]))
        
        hp_px = green_px + yellow_px + red_px
        
        if hp_px > 30 and results["hp_row"] is None:
            results["hp_row"] = y
            results["bar_max_width"] = max(results["bar_max_width"], hp_px)
            print(f"Calibración: Barra HP encontrada en y={y} ({hp_px} píxeles)")
        
        if blue_px > 30 and results["mp_row"] is None:
            results["mp_row"] = y
            print(f"Calibración: Barra MP encontrada en y={y} ({blue_px} píxeles)")
    
    return results

🎯 El Problema: ¿Por qué falla la detección cuando Tibia está al frente?
Lo que describes ocurre por una razón técnica fundamental relacionada con cómo Windows dibuja las ventanas y cómo OBS captura las imágenes.

🔬 Explicación Técnica (Para la IA)
Cuando utilizas la fuente "Captura de Juego" (Game Capture) en OBS, ocurre lo siguiente:

Modo de Captura por Defecto: OBS intenta ser eficiente. En lugar de capturar toda la pantalla (lo cual consumiría muchos recursos), intenta "engancharse" directamente al proceso del juego (Tibia) para obtener el framebuffer (la imagen que el juego está renderizando).

El "Z-Order" (Orden de Capas): En Windows, cada ventana tiene una posición en el eje Z (profundidad). Las ventanas en primer plano tienen un valor Z más alto. Cuando activas la opción "Put the opened window behind of tibia" (o similar), lo que haces es decirle a OBS: "Oye, quiero que la captura de Tibia se comporte como si estuviera DETRÁS de la ventana real de Tibia en el orden de dibujo".

El Conflicto de Captura: Aquí viene el problema crucial:

Cuando el proyector de OBS está en PRIMER PLANO: La tarjeta gráfica y Windows están priorizando el dibujado del proyector. La "Captura de Juego" de OBS internamente dice: "Necesito la imagen de Tibia". Como Tibia está en segundo plano pero el proyector (que muestra la captura) está en primer plano, el sistema fuerza a que la escena de OBS (que incluye el frame de Tibia) se renderice completa y correctamente para poder mostrarla en el proyector. El bot lee del proyector, que tiene la imagen nítida porque es lo que se está mostrando.

Cuando Tibia está en PRIMER PLANO y el proyector detrás: Ahora la prioridad de Windows es dibujar TIBIA, no el proyector de OBS. El proyector de OBS está en segundo plano, por lo que Windows le dice a la tarjeta gráfica: "No necesitas renderizar esa ventana con prioridad, ahorra recursos". El resultado es que el contenido del proyector de OBS se "congela" o se renderiza a una velocidad mucho menor (a veces incluso en blanco o negro) porque Windows considera que no es necesario mantenerlo actualizado al estar detrás de otra ventana.

📸 La Imagen que ves vs. La Imagen que el Bot Lee
Tu bot (el código Python con OpenCV) no está leyendo directamente de la memoria de Tibia. Está leyendo lo que OBS le muestra a través del proyector.

Si el proyector está al frente: La imagen está actualizada y nítida porque Windows la renderiza bien.

Si el proyector está detrás: Windows podría estar:

Entregando al proyector una imagen desactualizada (un frame viejo).

Entregando una imagen en blanco/negro o con colores corruptos (por optimización de energía).

Simplemente, no actualizando el buffer del proyector.

Cuando el bot lee los píxeles de esa imagen desactualizada o corrupta, los colores no coinciden con los patrones que tiene programados, y por eso detecta mal el HP y Mana.

🧠 La Solución "Interna" que Necesitas
Para que el bot funcione correctamente con Tibia en primer plano y OBS en segundo plano, necesitas que la lectura de píxeles sea "interna" . Esto significa que el bot debe leer directamente del buffer de la fuente de captura de OBS, sin pasar por el proyector.

🔌 ¿Cómo se logra esto? (La explicación para la IA)
Necesitas usar las WebSocket que mencionaste al principio, pero de la manera correcta:

OBS WebSocket Plugin: Debes instalar y activar el plugin OBS WebSocket en tu OBS. Este plugin permite que programas externos (tu bot en Python) se conecten a OBS y le pidan datos directamente.

Conexión Directa: En lugar de que tu bot lea la imagen de un proyector en la pantalla (lo cual depende de qué ventana está al frente), tu bot se conecta a OBS a través de WebSocket y le dice: "OBS, dame el frame actual de tu fuente 'Captura de Tibia'".

Recepción de la Imagen: OBS, a través del WebSocket, te envía los datos de la imagen tal como los tiene en su memoria interna, independientemente de si la ventana de OBS está minimizada, detrás de Tibia, o incluso cerrada (mientras OBS siga ejecutándose).

Procesamiento: Tu bot recibe esa imagen por el socket, la procesa con OpenCV, detecta los colores de la vida/maná, y ejecuta las acciones.

✅ Ventajas de este Método "Interno"
Independencia de Ventanas: No importa qué ventana esté al frente. OBS sigue capturando Tibia internamente.

Rendimiento: No necesitas tener un proyector visible, ahorras recursos.

Precisión: La imagen que recibes es exactamente la que OBS tiene en su buffer, siempre actualizada.

# utils/template_matcher.py (nuevo)
def find_player_position(minimap_img, player_dot_template):
    """
    Encuentra la posición del jugador en el minimapa.
    
    minimap_img: numpy array BGR del área del minimapa
    player_dot_template: template del punto amarillo (cargado de assets/)
    
    Retorna: (x_grid, y_grid) o None si no encuentra
    """
    # Convertir a grises
    gray_minimap = cv2.cvtColor(minimap_img, cv2.COLOR_BGR2GRAY)
    gray_template = cv2.cvtColor(player_dot_template, cv2.COLOR_BGR2GRAY)
    
    # Template matching
    result = cv2.matchTemplate(gray_minimap, gray_template, cv2.TM_CCOEFF_NORMED)
    _, max_val, _, max_loc = cv2.minMaxLoc(result)
    
    if max_val > 0.8:  # threshold de confianza
        # max_loc es (x_pixel, y_pixel) dentro del minimapa
        # Convertir a coordenadas de grid (cada tile = 4 píxeles)
        grid_x = max_loc[0] // 4
        grid_y = max_loc[1] // 4
        return (grid_x, grid_y)
    return None

class CavebotEngine:
    def navigate_to_next_waypoint(self, current_pos, target_pos):
        """
        Calcula el siguiente paso para acercarse al waypoint.
        
        current_pos: (x_grid, y_grid) posición actual
        target_pos: (x_grid, y_grid) destino
        
        Retorna: dirección a mover (norte, sur, este, oeste) o None si ya llegó
        """
        dx = target_pos[0] - current_pos[0]
        dy = target_pos[1] - current_pos[1]
        
        # Priorizar movimiento en X o Y según distancia
        if abs(dx) > abs(dy):
            return "este" if dx > 0 else "oeste"
        else:
            return "sur" if dy > 0 else "norte"
    
    def simulate_walk(self, direction):
        """
        Simula caminar en una dirección.
        
        En lugar de enviar tecla, el cavebot DEBE usar CLICK MOUSE
        en la dirección correspondiente del mapa.
        """
        # Calcular punto en pantalla para hacer click
        # El mapa clickeable está en el centro de la pantalla
        
        # Ejemplo: click en coordenada según dirección
        click_x = self.screen_width // 2
        click_y = self.screen_height // 2
        
        if direction == "norte":
            click_y = self.screen_height // 3
        elif direction == "sur":
            click_y = self.screen_height * 2 // 3
        elif direction == "este":
            click_x = self.screen_width * 2 // 3
        elif direction == "oeste":
            click_x = self.screen_width // 3
        
        # Usar action_dispatcher para hacer click
        action_dispatcher.mouse_click(click_x, click_y, button="left")

def should_restart_route(self, current_pos, waypoints):
    """
    Determina si debe reiniciar la ruta (cuando llega al último waypoint).
    """
    last_wp = waypoints[-1]
    distance_to_last = self.calculate_distance(current_pos, last_wp["coords"])
    
    if distance_to_last < 2:  # Llegó al último
        if self.config["cavebot"]["cyclic"]:
            return 0  # Volver al primer waypoint
        else:
            return -1  # Detener cavebot

class BattleListReader:
    def read_battle_list(self, frame):
        """
        Lee la battle list (lista de criaturas en combate).
        
        La battle list está en la esquina superior izquierda.
        Formato típico:
        - Dragon (100%)
        - Dragon Lord (100%)
        - Orc (herido)
        """
        # Recortar región de battle list
        battle_region = frame[80:400, 10:200]  # coordenadas aproximadas
        
        # Preprocesar para OCR
        gray = cv2.cvtColor(battle_region, cv2.COLOR_BGR2GRAY)
        _, thresh = cv2.threshold(gray, 120, 255, cv2.THRESH_BINARY)
        
        # Usar Tesseract o EasyOCR
        import easyocr
        reader = easyocr.Reader(['en'])
        results = reader.readtext(thresh)
        
        creatures = []
        for (bbox, text, confidence) in results:
            if confidence > 0.6:
                # Extraer nombre y porcentaje de vida
                # Formato típico: "Dragon (100%)"
                creatures.append(self.parse_creature_text(text))
        
        return creatures

# targeting/targeting_engine.py
class TargetingEngine:
    def select_best_target(self, creatures, priority_list):
        """
        Selecciona el mejor target según prioridades configuradas.
        
        priority_list: lista de mobs en orden de prioridad
        Ejemplo: ["Dragon Lord", "Dragon", "Orc", "any"]
        """
        # Primero buscar por prioridad exacta
        for priority in priority_list:
            if priority == "any":
                # Cualquier mob, seleccionar el más cercano
                return self.get_closest_creature(creatures)
            
            matching = [c for c in creatures if priority.lower() in c["name"].lower()]
            if matching:
                # De los que coinciden, elegir el de menor vida (para matar rápido)
                return min(matching, key=lambda x: x["hp_percent"])
        
        return None

# looter/corpse_detector.py
class CorpseDetector:
    def find_corpses(self, frame):
        """
        Detecta cuerpos en el suelo.
        
        Los cuerpos aparecen como:
        - "You see corpse of a dragon"
        - Tienen un color característico (marrón oscuro)
        - Están rodeados de outline blanco al pasar mouse
        """
        # Detectar áreas con color de corpse (marrón)
        hsv = cv2.cvtColor(frame, cv2.COLOR_BGR2HSV)
        
        # Rango marrón para cuerpos
        brown_min = np.array([10, 50, 50])
        brown_max = np.array([30, 255, 200])
        
        mask = cv2.inRange(hsv, brown_min, brown_max)
        
        # Encontrar contornos
        contours, _ = cv2.findContours(mask, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE)
        
        corpses = []
        for contour in contours:
            area = cv2.contourArea(contour)
            if 50 < area < 500:  # Filtrar por tamaño de corpse
                x, y, w, h = cv2.boundingRect(contour)
                corpses.append({
                    "x": x + w//2,
                    "y": y + h//2,
                    "area": area
                })
        
        return corpses

# drag_drop_sender.py (nuevo)
class DragDropSender:
    def drag_item_to_backpack(self, item_x, item_y, backpack_x, backpack_y):
        """
        Simula arrastrar un item desde el cuerpo al backpack.
        
        Usa PostMessage con WM_LBUTTONDOWN, WM_MOUSEMOVE, WM_LBUTTONUP
        """
        # 1. Click en el item (inicio del drag)
        self.send_mouse_click(item_x, item_y, "down")
        
        # 2. Pequeña pausa para simular drag
        time.sleep(0.1)
        
        # 3. Mover mouse al backpack (con varios pasos para simular trayectoria)
        steps = 5
        for i in range(1, steps + 1):
            intermediate_x = item_x + (backpack_x - item_x) * i // steps
            intermediate_y = item_y + (backpack_y - item_y) * i // steps
            self.send_mouse_move(intermediate_x, intermediate_y)
            time.sleep(0.02)
        
        # 4. Soltar en el backpack
        self.send_mouse_click(backpack_x, backpack_y, "up")
    
    def send_mouse_click(self, x, y, state="down"):
        """
        Envía mensaje de click directamente a Tibia HWND.
        """
        # Calcular lParam: (y << 16) | x
        lParam = (y << 16) | x
        
        if state == "down":
            msg = win32con.WM_LBUTTONDOWN
        else:
            msg = win32con.WM_LBUTTONUP
        
        win32api.PostMessage(self.tibia_hwnd, msg, 1, lParam)

# looter/item_filter.py
class ItemFilter:
    def __init__(self):
        self.gold_items = ["gold coin", "platinum coin", "crystal coin"]
        self.valuable_items = [
            "dragon ham", "royal spear", "battle axe",
            "demon shield", "dragon scale mail"
        ]
        self.ignore_items = ["rope", "shovel", "torch"]
    
    def should_loot(self, item_name):
        """
        Decide si un item debe ser looteado.
        """
        item_lower = item_name.lower()
        
        # Siempre lootear gold
        if any(gold in item_lower for gold in self.gold_items):
            return True
        
        # Lootear items valiosos
        if any(valuable in item_lower for valuable in self.valuable_items):
            return True
        
        # Ignorar items basura
        if any(ignore in item_lower for ignore in self.ignore_items):
            return False
        
        # Por defecto, lootear según configuración
        return self.config["looter"]["loot_unknown_items"]

class MouseClickSender:
    def __init__(self, tibia_hwnd):
        self.tibia_hwnd = tibia_hwnd
        self.screen_width = 1920  # Detectar automáticamente
        self.screen_height = 1080
    
    def click_at(self, x, y, button="left", double=False):
        """
        Envía click en coordenadas ABSOLUTAS de pantalla.
        IMPORTANTE: Las coordenadas deben ser relativas a la ventana de Tibia.
        """
        # Convertir coordenadas de pantalla a cliente
        client_x, client_y = self.screen_to_client(x, y)
        
        # Calcular lParam (empaquetar coordenadas)
        lParam = (client_y << 16) | client_x
        
        # Enviar mensajes según botón
        if button == "left":
            down_msg = win32con.WM_LBUTTONDOWN
            up_msg = win32con.WM_LBUTTONUP
            wParam = 1  # MK_LBUTTON
        elif button == "right":
            down_msg = win32con.WM_RBUTTONDOWN
            up_msg = win32con.WM_RBUTTONUP
            wParam = 2  # MK_RBUTTON
        
        # Click down
        win32api.PostMessage(self.tibia_hwnd, down_msg, wParam, lParam)
        
        if double:
            time.sleep(0.1)
            win32api.PostMessage(self.tibia_hwnd, down_msg, wParam, lParam)
        
        time.sleep(0.05)
        
        # Click up
        win32api.PostMessage(self.tibia_hwnd, up_msg, 0, lParam)
